471: // ═══════ ANIMATED COUNT-UP ═══════
472: function animateCountUp(element, target, duration = 1200, prefix = '', suffix = '', decimals = 0) {
473:     const startTime = performance.now();
474:     const startVal = 0;
475:     
476:     function update(currentTime) {
477:         const elapsed = currentTime - startTime;
478:         const progress = Math.min(elapsed / duration, 1);
479:         // Ease-out cubic for a satisfying deceleration
480:         const eased = 1 - Math.pow(1 - progress, 3);
481:         const current = startVal + (target - startVal) * eased;
482:         
483:         if (decimals === 0) {
484:             element.textContent = prefix + Math.round(current).toLocaleString() + suffix;
485:         } else {
486:             element.textContent = prefix + current.toFixed(decimals) + suffix;
487:         }
488:         
489:         if (progress < 1) {
490:             requestAnimationFrame(update);
491:         }
492:     }
493:     requestAnimationFrame(update);
494: }
